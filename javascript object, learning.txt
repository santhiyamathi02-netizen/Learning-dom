//Problem: Number Analyzer Write a function analyzeNumbers(nums) that:  Takes an array of numbers  Finds: sum of all numbers Count of even numbers
    function analyzenumbers(nums)
{
    let sum = 0;
    let evencount = 0;

    for (let i = 0; i < nums.length; i++)
    {
        sum += nums[i];

        if (nums[i] % 2 == 0)
        {
            evencount++;
        }
    }

        evencount: evencount
    };
}
