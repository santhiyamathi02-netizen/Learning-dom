//Assignment 7: Factorial Calculator
function calculator(factorial)
{
  if (typeof factorial !== "number")
      return "Enter a valid number";
  
  let result =1;
  for(let i=1; i <= factorial; i++)
  {
      result *=i;
  }
  return result;
}
console.log(calculator(5));

}

//Assignment 9: Fibonacci Sequence
function fibonacci(n)
{
    if(typeof n !== "number" || n <= 0)
    {
    return "Enter a valid number"
}
if(n ==1 || n==2)
{
    return 1;
}
let a =1;
let b =1;
let next;
for(let i=3; i <=n; i++)
{
    next= a+b;
    a=b;
    b=next;
}
return b;
}
console.log(fibonacci(6));
