function calculate(factorialnumber)
{
    if(typeof factorialnumber !=="number")
    {
        return "error message enter a valid number"
    }

{
    let result = 1;
    for(let a=1; a <= factorialnumber; a++)

}

}
}

//Write a function sumOfDigits(num) that returns the sum of all digits of the number.
function sumofdigits(num1) {
    if (typeof num1 !== "number") {
        return "error message enter a valid number";
    }

    let str = num1.toString();
    let value = 0;

    for (let u = 0; u < str.length; u++) {
        value += Number(str[u]);
    }

    return value;
}

console.log(sumofdigits(456));
