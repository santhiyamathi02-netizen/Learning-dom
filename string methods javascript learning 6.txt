//Write a function that checks whether a given string is a palindrome. A palindrome reads the same forwards and backwards (example: "madam"). Convert the string to lowercase, reverse it, and compare both versions. If they match, return true; otherwise return false.
function palindrome(string)
{
    let str = string.toLowerCase();

    let reverse = str.split("").reverse().join("")
    return string === reverse;
}
console.log(palindrome("madam"));

//Create a function that finds the longest word in a sentence. Split the sentence into an array of words, loop through them, and compare lengths. Return the word with the greatest number of characters.
function longestword(sentence)
{
    let words = sentence.split(" ");
    let longest = words[0];
    for(let i=0; i < words.length; i++)
    {
        if(words[i].length > longest.length)
        {
            longest = words[i];
        }
    }
    return longest;
}
console.log(longestword("I am the Developer"));
